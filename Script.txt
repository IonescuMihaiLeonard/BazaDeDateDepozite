(

        CONSTRAINT ISTORIC_JOB_ANGAJAT_ID_NN NOT NULL
        CONSTRAINT ISTORIC_JOB_ANGAJAT_ID_CK CHECK (ANGAJAT_ID > 0),

    DATA_INCEPERE DATE
        CONSTRAINT ISTORIC_JOB_DATA_INCEPERE_NN NOT NULL,

    CONSTRAINT ISTORIC_JOB_ANGAJAT_FK FOREIGN KEY(ANGAJAT_ID)
    REFERENCES ANGAJATI(ANGAJAT_ID) ON DELETE CASCADE
